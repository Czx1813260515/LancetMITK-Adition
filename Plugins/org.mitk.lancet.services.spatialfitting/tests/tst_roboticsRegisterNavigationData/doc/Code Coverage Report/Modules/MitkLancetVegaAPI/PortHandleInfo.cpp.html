#include <iomanip>
#include <sstream>

#include "PortHandleInfo.h"

PortHandleInfo::PortHandleInfo(std::string portHandle, uint8_t status)
{
	portHandle_ = portHandle;
	toolType_ = "";
	toolId_ = "";
	revision_ = "";
	serialNumber_ = "";
	status_ = status;
}

PortHandleInfo::PortHandleInfo(std::string portHandle, std::string toolType,
		        std::string toolId, std::string revision,
				std::string serialNumber, uint8_t status)
{
	portHandle_ = portHandle;
	toolType_ = toolType;
	toolId_ = toolId;
	revision_ = revision;
	serialNumber_ = serialNumber;
	status_ = status;
}


std::string PortHandleInfo::getPortHandle() const
{
	return portHandle_;
}

std::string PortHandleInfo::getToolId() const
{
	return toolId_;
}

std::string PortHandleInfo::getRevision() const
{
	return revision_;
}

std::string PortHandleInfo::getSerialNumber() const
{
	return serialNumber_;
}

std::string PortHandleInfo::getStatus() const
{
	std::stringstream stream;
	stream << ((status_ & ToolInPort) ? "ToolInPort|" : "")
		   << ((status_ & Switch1Closed) ? "Switch1Closed|" : "")
		   << ((status_ & Switch2Closed) ? "Switch2Closed|" : "")
		   << ((status_ & Switch3Closed) ? "Switch3Closed|" : "")
		   << ((status_ & PortInitialized) ? "PortInitialized|" : "")
		   << ((status_ & PortEnabled) ? "PortEnabled|" : "")
		   << ((status_ & CurrentSensed) ? "CurrentSensed" : "");
	std::string retVal = stream.str();
	if (retVal.at(retVal.size() - 1) == '|')
	{
		retVal.erase(retVal.end() - 1);
	}
	return retVal;
}

std::string PortHandleInfo::toString() const
{
	std::stringstream stream;
	stream << std::hex << std::setfill('0');
	stream << "portHandle[" << getPortHandle() << "].status=" << getStatus();
	return stream.str();
}
