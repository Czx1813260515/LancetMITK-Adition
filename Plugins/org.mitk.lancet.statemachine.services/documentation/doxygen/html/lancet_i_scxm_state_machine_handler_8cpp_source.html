165 void IScxmStateMachineHandler::onScxmlStateMachine_stateChange(bool active)
166 {
167 if (false == active)
168 {
169 this->SetLastActiveStateMachine(this->GetActiveStateMachine());
170 emit this->StateExit(this->GetLastActiveStateMachine().GetPointer());
171 return;
172 }
173
174 if (this->GetAssignor().IsNotNull() && !this->GetAssignor()->GetQtStateMachine().isNull())
175 {
176 auto stateMachineService = IScxmlStateMachineService::GetScxmlStateMachineService();
177 if (stateMachineService.IsNull())
178 {
179 // Do nothing
180 return;
181 }
182
183 // find active state and update it.
184 for (auto substate : this->GetAssignor()->GetSubStateMachines())
185 {
186 bool is_active = this->GetAssignor()->GetQtStateMachine()->activeStateNames()
187 .contains(substate->GetStateId().toLower());
188 if (active && substate.IsNotNull() && is_active)
189 {
190 this->SetActiveStateMachine(substate);
191 emit this->StateEnter(this->GetActiveStateMachine().GetPointer());
192 break;
193 }
194 }
195 }
196 }
